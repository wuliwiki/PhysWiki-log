\begin{theorem}{Gauss 二次互反律}
若 $p$ 与 $q$ 都是奇素数，则
\begin{equation}
\left(\frac pq\right)\left(\frac qp\right) = (-1)^{p' q'} ~,
\end{equation}
其中 $p' = (p-1)/2$ 而 $q' = (q-1)/2$。
\end{theorem}
