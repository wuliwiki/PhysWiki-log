% 电抗 容抗 感抗

\pentry{电容\upref{Cpctor}， 阻抗\ 电抗\upref{impeda}}

\footnote{本文参考 Wikipedia \href{https://en.wikipedia.org/wiki/Electrical_reactance}{相关页面}．}我们已经知道阻抗的虚部称为\textbf{电抗（reactance）}， 实部称为电阻． 理想的电阻器只有电阻没有电抗（\autoref{impeda_ex1}~\upref{impeda}）， 而理想的电容器和电感器只有电抗没有电阻． 换言之， 电容或电感电压和电流总是保持 $\pi/2$ 的相位差． 我们把电容器和电感器的电抗分别称为\textbf{容抗（capacitive reactance）}和\textbf{感抗（inductive reactance）}．

\subsection{容抗}
根据电容公式\autoref{Cpctor_eq3}~\upref{Cpctor} （$I = C\dv*{U}{t}$）． 令交流电为 $\E^{-\I \omega t}$
\begin{equation}
I = -\I \omega CV
\end{equation}
电容的阻抗为
\begin{equation}
Z_C = \frac{V}{I} = \frac{\I}{\omega C}
\end{equation}
而容抗为
\begin{equation}
X_c = \Im{Z_c} = \frac{1}{\omega C}
\end{equation}
所以有类似欧姆定律的关系
\begin{equation}
V = Z_c I = \I X_c I
\end{equation}

\subsection{感抗}
根据电感公式\autoref{Induct_eq2}~\upref{Induct}（$V = L\dv*{I}{t}$）． 令交流电为 $\E^{-\I \omega t}$
\begin{equation}
V = -\I\omega L I
\end{equation}
\begin{equation}
Z_L = \frac{V}{I} = -\I\omega L
\end{equation}
\begin{equation}
X_L = -\omega L
\end{equation}
注意和容抗的符号相反．
