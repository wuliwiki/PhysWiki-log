    \[    H = \frac{p^2}{2\mu} + V(\vec{r})~ \]
    \begin{enumerate}
        \item 对于定态，证明位力(virial)定理： $2\langle \overline T \rangle = \langle \vec{r} \cdot \nabla V(\vec{r}) \rangle$
        \item 若势场为中心力场 $V(r) = -\frac{k}{r^s}$ ($k>0$)，证明存在束缚态的条件为 $0 < s < 2$。
    \end{enumerate}
