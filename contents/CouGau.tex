% 库仑规范（量子）

\pentry{电磁场中的单粒子薛定谔方程\upref{QMEM}}

库仑规范\upref{Cgauge}推导起来较为简单， 用角标 $C$ 表示库仑规范， 则矢势满足 $\div \bvec A_C = 0$， 标势 $\varphi_C$ 满足（\autoref{Cgauge_eq1}~\upref{Cgauge}）
\begin{equation}
\varphi(\bvec r, t) = \frac{1}{4\pi\epsilon_0} \int \frac{\rho(\bvec r', t)}{\abs{\bvec r - \bvec r'}} \dd[3]{r'}
\end{equation}
结合矢量算符法则\autoref{VopEq_eq1}~\upref{VopEq} 得
\begin{equation}
\div (\bvec A_C \Psi) = (\div \bvec A_C) \Psi + \bvec A_C \vdot (\grad \Psi) = \bvec A_C \vdot (\grad \Psi)
\end{equation}
所以\autoref{QMEM_eq2}~\upref{QMEM} 简化为
\begin{equation}\label{CouGau_eq4}
H = -\frac{1}{2m} \laplacian + \I \frac{q}{m} \bvec A_C \vdot \Nabla + \frac{q^2}{2m} \bvec A_C^2 + q\varphi_C
\end{equation}
