\pentry{同余与剩余类\nref{nod_modres}}{nod_3b36}
\begin{definition}{完系求和记号}
若变量 $h$ 遍历某整数 $n$ 的一个完全剩余系，对 $f(h)$ 求和，则可以用记号 $h(n)$ 表示求和范围，即
\begin{equation}
\sum_{h(n)} f(h) ~.
\end{equation}
\end{definition}

\begin{definition}{缩系求和记号}
若变量 $h$ 变量某整数 $n$ 的一个缩系，对 $g(h)$ 求和，则可以用记号 $h^*(n)$ 表示求和范围，即
\begin{equation}
\sum_{h^*(n)} g(n) ~.
\end{equation}

\end{definition}
