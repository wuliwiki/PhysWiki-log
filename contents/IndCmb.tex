\pentry{电感\upref{Induct}}{nod_5a1c}

串联：
\begin{equation}
L = L_1 + L_2~.
\end{equation}


并联：
\begin{equation}
\frac{1}{L} = \frac{1}{L_1} + \frac{1}{L_2}
\quad \text{或} \quad
L = \frac{L_1L_2}{L_1 + L_2}~.
\end{equation}
