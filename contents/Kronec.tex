\textbf{克罗内克 $\delta$ 函数（Kronecker delta function）}， 定义为
\begin{equation}\label{eq_Kronec_2}
\delta_{ij} =
\begin{cases}
1 & (i = j)\\
0 & (i \ne j)~.
\end{cases}
\end{equation}
易证 $\delta$ 函数的两个常用性质
\begin{equation}\label{eq_Kronec_1}
\sum_j a_j \delta_{ij} = a_i~,
\end{equation}
\begin{equation}
\sum_{ij} a_i b_j \delta_{ij} = \sum_k a_k b_k~.
\end{equation}
